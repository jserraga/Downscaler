    def main():
    from T1_230313_YearParsing import *
    import argparse
    import sys
    import pandas as pd
    import sklearn.metrics as sk
    import numpy as np
    import os

    os.system('cls||clear')

    int_year = pd.to_numeric(sys.argv[1:][0])
    np.set_printoptions(linewidth=160)

    if (int_year != 2017 and int_year != 2018 and int_year != 2019 and int_year != 2020 and int_year != 2021 and int_year != 2022):
        print("This script only accepts the following values as argument: 2019, 2020, 2021, 2022")
    else:
        print("Process started. Please wait...")

        df_RPA = pd.read_csv("RPA_2p_90p_" + str(int_year) + ".csv")
        df_LUS_exc = year_parsing(int_year)

        #Selecting max 
        df_LUS_max = df_LUS_exc.loc[df_LUS_exc.groupby("NGC_NEW", sort = False)["SUM_AREA_CLAIMED_HA"].idxmax()][["NGC_NEW","LAND_USE_CODE_NEW","SUM_AREA_CLAIMED_HA"]]
        # filter out ioc
        df_RPA = df_RPA[df_RPA.RPA_2plus_IoC >= 0.9]
        df_RPA = df_RPA[df_RPA.RPA_2plus_IoC <= 1.1]

        # inner join - select only those that are in both. First rename field names for join
        df_RPA_renamed = df_RPA.rename(columns = {"RPA_2plus_NGC":"NGC_NEW"})
        df_RPA_LUS_J = pd.merge(df_RPA_renamed, df_LUS_max, how = "inner", on = "NGC_NEW")

        print("Calculating Cohen's Kappa for the " + str(int_year) + " dataset")
        y_true = df_RPA_LUS_J["LAND_USE_CODE_NEW"]
        y_pred = df_RPA_LUS_J["crop" + str(int_year)]
        c_m = sk.confusion_matrix(y_true,y_pred, labels = df_RPA_LUS_J["LAND_USE_CODE_NEW"].unique())

        print("The Coehn's Kappa value for the " + str(int_year) + " dataset is " + str(round(sk.cohen_kappa_score(y_true,y_pred),4)))
        print("Accuracy score: " + str(sk.accuracy_score(y_true,y_pred)))
        print(sk.classification_report(y_true,y_pred, zero_division = 0))
        print("Calculating confusion matrix...")
        df_ct = pd.crosstab(y_true, y_pred)
        print(df_ct)
        print("Exporting...")
        df_ct.to_csv("confusion_matrix_" + str(int_year) + ".csv")
        print("The confusion matrix has been saved in the local drive.")
        print("Completed.")
